180 	var AutoBindProto = {
181 		/** @private */
182 		blackList: ["initialize", "parent", "$constructor", "addEventListener", "removeEventListener" ]
183 		/**
184 		 * AutoBind the application.
185 		 * @name autobind
186 		 * @methodOf soma.core.AutoBind#
187 		 * @example
188 MyAutoBoundClass = new Class({
189 	Implements: soma.core.AutoBind,
190 	initialize: function() {
191 		this.autobind();
192 	}
193  });
194 
195 		 */
196 		,autobind: function() {
197 			if (this.wasAutoBound) {
198 				return;
199 			}
200 			var o = this;
201 			var ab = o["AutoBindPattern"];
202 			var coreAb = "([lL]istener|[hH]andler|[cB]allback)$";
203 			if (!ab) {
204 				ab = coreAb;
205 			} else {
206 				ab = coreAb + "|" + ab;
207 			}
208 			for (var k in o) {
209 				if (typeof o[k] == "function") {
210 					if (this._autobindIsBlacklisted(k)) {
211 						continue;
212 					}
213 					if (!k.match(ab)) {
214 						continue;
215 					}
216 					o[k] = o[k].bind(o);
217 				}
218 			}
219 		}
220 		/** @private */
221 		,_autobindIsBlacklisted: function(name) {
222 			var bl = this.blackList;
223 			for (var i = 0; i < bl.length; i++) {
224 				if (bl[i] == name) {
225 					return true;
226 				}
227 			}
228 			return false;
229 		}
