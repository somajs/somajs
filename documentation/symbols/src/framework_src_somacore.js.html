181 	/**
182 	 * 
183 	 * provides the functionality to autobind with implicit need to keep object scope like event listeners and handlers/callbacks
184 	 * ending with *Listener or *Handler
185 	 * Wires and Mediators are implementing instance scope autobinding upon registration
186 	 */
187 	var AutoBindProto = {
188 		blackList: ["initialize", "parent", "$constructor", "addEventListener", "removeEventListener" ]
189 		,autobind: function() {
190 			if (this.wasAutoBound) {
191 				return;
192 			}
193 			var o = this;
194 			var ab = o["AutoBindPattern"];
195 			var coreAb = "([lL]istener|[hH]andler|[cB]allback)$";
196 			if (!ab) {
197 				ab = coreAb;
198 			} else {
199 				ab = coreAb + "|" + ab;
200 			}
201 			for (var k in o) {
202 				if (typeof o[k] == "function") {
203 					if (this._autobindIsBlacklisted(k)) {
204 						continue;
205 					}
206 					if (!k.match(ab)) {
207 						continue;
208 					}
209 					o[k] = o[k].bind(o);
210 				}
211 			}
212 		}
213 		,_autobindIsBlacklisted: function(name) {
214 			var bl = this.blackList;
215 			for (var i = 0; i < bl.length; i++) {
216 				if (bl[i] == name) {
217 					return true;
218 				}
219 			}
220 			return false;
221 		}
222 	};
